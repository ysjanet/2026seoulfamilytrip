        async function syncSiteData(id) {

            const site = sites.find(s => s.id === id);

            if(!site) return;


            const card = document.getElementById(`card-${id}`);

            if(card) card.querySelector('.loading-overlay').style.display = 'flex';


            const systemPrompt = `你是一個專業首爾導遊。請針對使用者提供的「景點或餐廳名稱」，搜尋 Google Maps 與韓國在地資訊。

            {

                "address": "中文完整地址",

            }`;

            try {

                if(apiKey) {

                    const response = await fetch(`https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash-preview-09-2025:generateContent?key=${apiKey}`, {

                        method: 'POST',

                        headers: { 'Content-Type': 'application/json' },

                        body: JSON.stringify({

                            contents: [{ parts: [{ text: userQuery }] }],

                            systemInstruction: { parts: [{ text: systemPrompt }] },

                            generationConfig: { responseMimeType: "application/json" },

                            tools: [{ "google_search": {} }]

                        })

                    });

                    const result = await response.json();

                    if(result.candidates && result.candidates[0].content.parts[0].text) {

                        const info = JSON.parse(result.candidates[0].content.parts[0].text);

                        site.address = info.address;

                        site.hours = info.hours;

                        site.elevator = info.elevator;

                    }

                }

                

                // 修正 1：換成不會失效的 Placehold 圖片佔位符

                saveSites();

                renderItinerary();

            } catch (error) {

                console.error("Sync Error:", error);

                if(card) card.querySelector('.loading-overlay').style.display = 'none';

            }

        }

        async function updateLiveExchangeRate() {

            try {

                const res = await fetch('https://api.exchangerate-api.com/v4/latest/HKD');

                const data = await res.json();

                if(data.rates.KRW) {

                    exchangeRate = data.rates.KRW;

                    document.getElementById('rate-display').innerText = exchangeRate.toFixed(1);

                    // 修正 2：更新即時匯率後，強制重繪算好的總金額

                    renderExpenses();

                }

            } catch(e) {}

        }

        function convertCurrency() {

            const val = document.getElementById('krw-input').value;

            document.getElementById('hkd-result').innerText = val ? (val / exchangeRate).toFixed(2) : "0.00";

        }

        }

                return `<div class="flex justify-between items-center bg-white p-4 rounded-2xl shadow-sm border border-gray-50 mb-2">

                    <div class="font-bold text-sm text-gray-700">${e.name} <span class="text-[10px] text-gray-300 ml-2">₩${Number(e.amt).toLocaleString()}</span></div>

                    <button onclick="deleteExpense(${e.id})" class="text-gray-200"><i class="fa-solid fa-circle-xmark"></i></button>

                </div>`;

            }).join('');

            document.getElementById('total-expense').innerText = "$" + (total / exchangeRate).toFixed(2);

        }

        function deleteExpense(id) { expenses = expenses.filter(e => e.id !== id); saveExpenses(); renderExpenses(); }

        function addShopItem() {

                    <span class="font-bold text-sm flex-1 ${s.done ? 'line-through text-gray-300' : 'text-gray-600'}">${s.text}</span>

                    <button onclick="deleteShop(${s.id})" class="text-gray-200"><i class="fa-solid fa-trash-can"></i></button>

                </div>

            `).join('');

        }

        function deleteShop(id) { shopping = shopping.filter(s => s.id !== id); saveShop(); renderShopping(); }

        async function saveAndSync() {

            const title = document.getElementById('site-title').value;

            const day = document.getElementById('site-day').value;

            const time = document.getElementById('site-time').value;

            if(!title || !time) return;

            const newId = Date.now();

            sites.push({ id: newId, day, time, title, address: "", hours: "", elevator: "", img: "" });

            

            // 修正 3：新增行程後，自動將主畫面切換到該天數

            setDay(day);

            

            saveSites(); closeModal(); renderItinerary();

            await syncSiteData(newId);

        }

        async function syncAllData() { 

            const daySites = sites.filter(s => s.day === currentDay);

            for(const s of daySites) {

                await syncSiteData(s.id);

            }

        }
